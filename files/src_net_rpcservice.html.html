DIJS.bind('pb.net.RpcService.__ctor__', require => {

  const $ = require('pb.JQuery');
  const normalize = require('pb.net.Utils').normalize;

  const __get__ = Symbol('get');
  const __io__ = Symbol();
  const __onUpdate__ = Symbol();
  const __remove__ = Symbol('remove');
  const __set__ = Symbol('set');
  const __triggerUpdates__ = Symbol('triggerUpdates');
  const __updateAncestorIndexes__ = Symbol('updateAncestorIndexes');

  /**
   * Wrapper class for net.IO.
   *
   * @class net.RpcService
   * @since 3.1.0
   */
  class Rpc {
    /**
     * @constructor
     * @class net.RpcService
     * @param {net.IO} io Implementation of {{#crossLink "net.AbstractIO"}}{{/crossLink}}.
     */
    constructor(io) {
      this[__io__] = io;

      io.listen(this[__onUpdate__].bind(this));
    }

    /**
     * Retrieves the value in the given path, constructing an object if necessary.
     *
     * @method __get__
     * @param {string} path The path whose value should be returned.
     * @return {Promise} [description]
     */
    [__get__](path) {
      return this[__io__].get(path)
          .then(data => {
            if (data && data.$index) {
              let promises = [];
              for (let key of data.keys) {
                promises.push(this[__get__](normalize(path, key))
                  .then((key, value) => {
                    return [key, value];
                  }.bind(null, key)));
              }
              return Promise.all(promises)
                  .then(pairs => {
                    let object = {};
                    for (let pair of pairs) {
                      let [key, value] = pair;
                      object[key] = value;
                    }
                    return object;
                  });
            } else {
              return Promise.resolve(data);
            }
          });
    }

    /**
     * Handler called when there is an update from the server.
     *
     * @method __onUpdate__
     * @param {string} path Path of resource that is updated.
     * @param {Any} newValue The new updated value.
     * @private
     */
    [__onUpdate__](path, newValue) {
      let normalizedPath = normalize(path);
      this[__triggerUpdates__](normalizedPath);
    }

    /**
     * Removes the entry in the given path, and update the ancestor indexes.
     *
     * @method __remove__
     * @param {string} path Path whose entry should be deleted.
     * @return {promise} Promise that will be resolved when the removal is done.
     */
    [__remove__](path) {
      let parts = path.split('/');
      let id = parts.pop();

      let promises = [this[__io__].set(path, null)];
      if (parts.length > 1) {
        let parentPath = parts.join('/');
        promises.push(this[__io__].get(parentPath)
            .then(value => {
              if (value && value.$index && value.keys.indexOf(id) >= 0) {
                value.keys.splice(value.keys.indexOf(id), 1);
              }

              if (value.keys.length === 0) {
                return this[__remove__](parentPath);
              } else {
                return this[__io__].set(parentPath, value);
              }
            }));
      }

      return Promise.all(promises);
    }

    /**
     * Sets the given data to the given path.
     *
     * @method __set__
     * @param {string} path Path to store the data in.
     * @param {Any} data Data to be stored. This should be a primitive, Object, or Array.
     * @return {Promise} Promise that will be resolved when the set operation is completed.
     * @private
     */
    [__set__](path, data) {
      let promises = [];
      if (data instanceof Object && !(data instanceof Array)) {
        promises.push(this[__io__].set(
            path,
            {
              $index: true,
              keys: Object.keys(data)
            }));
        for (let key in data) {
          promises.push(this[__set__](normalize(path, key), data[key]));
        }
      } else {
        promises.push(this[__io__].set(path, data));
      }
      return Promise.all(promises);
    }

    /**
     * Triggers updates on the given path, and all the ancestor paths.
     *
     * @method __triggerUpdates__
     * @param {string} path The path to trigger updates on.
     * @private
     */
    [__triggerUpdates__](path) {
      let parts = path.split('/');
      while (parts.length > 1) {
        let triggerPath = parts.join('/');
        if (triggerPath.length > 0) {
          $(this).trigger(parts.join('/'));
        }
        parts.pop();
      }
    }


    /**
     * Updates the ancestors in the given path to contain indexes.
     *
     * @method __updateAncestorIndexes__
     * @param {string} path Ancestor path to update.
     * @return {Promise} Promise that will be resolved when the update operation is completed.
     * @private
     */
    [__updateAncestorIndexes__](path) {
      let ancestorPromises = [];
      let parts = path.split('/');
      let ancestorPath = normalize(parts.splice(0, 1)[0]);

      for (let part of parts) {
        ancestorPromises.push(this[__io__].get(ancestorPath)
            .then((path, part, ancestor) => {
              if (!ancestor || !ancestor.$index) {
                ancestor = { $index: true, keys: [] };
              }

              if (ancestor.keys.indexOf(part) < 0) {
                ancestor.keys.push(part);
              }

              // Don't use __set__ since we do not want to unravel the array.
              return this[__io__].set(path, ancestor);
            }.bind(null, ancestorPath, part)));
        ancestorPath = normalize(ancestorPath, part);
      }

      return Promise.all(ancestorPromises);
    }

    /**
     * Adds a new object on the given path.
     *
     * @method add
     * @param {string} path Path to add a new object to.
     * @return {Promise} Promise that will be resolved with the new object's ID when addition is
     *    successful.
     */
    add(path) {
      let normalizedPath = normalize(path);
      return this[__updateAncestorIndexes__](path)
          .then(() => this[__io__].get(normalizedPath))
          .then(index => {
            if (!index || !index.$index) {
              index = {
                $index: true,
                keys: []
              };
            }

            // Find the first non existent key
            let newId = 0;
            while (index.keys.indexOf(`${newId}`) >= 0) {
              newId++;
            }

            return this.set(normalize(path, `${newId}`), {})
                .then(() => `${newId}`);
          });
    }

    /**
     * Removes the object in the given path.
     *
     * @method remove
     * @param {string} path Path of the object to be removed.
     * @return {Promise} Promise that will be resolved when the object has been deleted.
     */
    remove(path) {
      return this[__remove__](normalize(path));
    }

    /**
     * Retrieves the value corresponding to the given path. If the path points to an index, this
     * will compile all the paths into an object.
     *
     * @method get
     * @param {string} path The path whose value should be retrieved.
     * @return {Promise} Promise that will be resolved with the value corresponding to the given
     *    path.
     */
    get(path) {
      return this[__get__](normalize(path));
    }

    /**
     * Sets the value of the given path. If the data is a non array object, this will create
     * an index object at the given path. This method will then store the object's keys in
     * individual entries. For example, suppose we store the following object in '/a':
     *
     * ```json
     * {
     *   b: 1,
     *   c: {
     *     d: 2
     *   }
     * }
     * ```
     *
     * This will result in the following mapping:
     *
     * ```
     * /a       -> (index)
     * /a/b     -> 1
     * /a/b/c   -> (index)
     * /a/b/c/d -> 2
     * ```
     *
     * @method set
     * @param {string} path The path to set the value to.
     * @param {Any} data Data to set the path's value to. This should be a primitive, Object, or
     *    Array.
     * @return {Promise} Promise that will be resolved when setting the value is successful.
     */
    set(path, data) {
      let normalizedPath = normalize(path);
      return this[__updateAncestorIndexes__](normalizedPath)
          .then(() => {
            return this[__set__](normalizedPath, data);
          });
    }
  }

  return Rpc;
})
.bind('pb.net.RpcService', require => {
  const IO = require('pb.net.IO');
  const Rpc = require('pb.net.RpcService.__ctor__');
  return new Rpc(IO);
});
</script>
